function isAcceptablePassword(password: string): boolean {

    const cond1 = password.length > 6;
    var cond2 = /\d/.test(password);
    var cond3 = /\D/.test(password);
    if (password.length > 9) cond2 = cond3 = true; // cond 4
    const cond5 = !(/password/i.test(password));
    const cond6 = new Set(password.split("")).size > 2;
    
    return !([cond1, cond2, cond3, cond5, cond6].includes(false));
}
